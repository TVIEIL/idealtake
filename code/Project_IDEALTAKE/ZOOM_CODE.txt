Private Sub ZOOM_MOIN_webbrowser1_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles ZoomMoinToolStripMenuItem4.Click, ToolStripMenuItem7.Click

    zoom1 = zoom1 - 10

    If (zoom1 > 10) Then
        Zoom_webbrowser(zoom1, "WebBrowser1")
    End If

    If (zoom1 <= 10) Then
        zoom1 = 10
        Zoom_webbrowser(zoom1, "WebBrowser1")
    End If

    ToolStripMenuItem5.Text = "Zoom " & zoom1.ToString & "%"

End Sub
